% ----- Front page 
\makeatletter
\renewcommand{\and}{, } % for the authors ... 
\def\frontpage{
	\thispagestyle{empty}
	\newgeometry{margin=0pt}
	%\setlength{\unitlength}{1in}
	\begin{picture}(0,0)
		\put(-4pt,-90pt){
			\begin{minipage}{\paperwidth}
				\noindent
				\begin{flushleft}
				\includegraphics*[width=\paperwidth]{skeleton/EC_banner.pdf}
				\end{flushleft}
			\end{minipage}
			}
		\put(0.1\paperwidth, -0.35\paperheight){
			\parbox{0.75\paperwidth}{
				\begin{flushleft}
				\sffamily\Huge\@title
				\end{flushleft}
			}		
		}
		\put(0.1\paperwidth, -0.45\paperheight){
			\parbox{0.75\paperwidth}{
				\begin{flushleft}
				\sffamily\Large\color{BlueTitle}\subtitle
				\end{flushleft}
			}		
		}
		\put(0.5\paperwidth, -0.6\paperheight){
			\parbox{.4\paperwidth}{
				\sffamily\large {\@author}
			}
		}
		\put(0.5\paperwidth, -0.7\paperheight){
			\parbox{.4\paperwidth}{
				\sffamily\large\textbf {\@date}
			}
		}
		\put(-4pt,-\paperheight){\FrontFigure}
	\end{picture}
	\vfill
	\begin{picture}(1,0.1)
		\put(0.45\paperwidth, 0){
			\includegraphics{skeleton/BottomRibbon.pdf}
			}
	\end{picture}
	\restoregeometry
}
\makeatother
